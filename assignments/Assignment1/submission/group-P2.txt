CREATE TABLE Bookings(
	seats   INTEGER,
	bno     CHAR(16), 
	cost    REAL,
	isFor   CHAR(16)  NOT NULL,
	dropOff CHAR(16),
	pickup  CHAR(16),
	reserve CHAR(320) NOT NULL,
	PRIMARY KEY (bno),
	FOREIGN KEY (isFor)   REFERENCES Rides,
	FOREIGN KEY (dropOff) REFERENCES Locations,
	FOREIGN KEY (pickup)  REFERENCES Locations,
	FOREIGN KEY (reserve) REFERENCES Members
);

CREATE TABLE Members(
	email CHAR(320),
	name  CHAR(64),
	phone CHAR(16),
	PRIMARY KEY (email)
);

CREATE TABLE Reviews(
	rid      CHAR(16),
	rdate    DATE,
	rtext    CHAR(320),
	rscore   INTEGER,
	reviewer CHAR(320) NOT NULL,
	reviewee CHAR(320) NOT NULL,
	assoc    CHAR(16),
	PRIMARY KEY (rid),
	FOREIGN KEY (reviewer) REFERENCES Members,
	FOREIGN KEY (reviewee) REFERENCES Members,
	FOREIGN KEY (assoc)    REFERENCES Rides
);

CREATE TABLE Rides(
	rno     CHAR(16),
	price   REAL,
	rdate   DATE,
	seats   INTEGER,
	lugDesc CHAR(1024),
	src     CHAR(32)  NOT NULL,
	dst     CHAR(32)  NOT NULL,
	offer   CHAR(320) NOT NULL,
	assoc   CHAR(16),
	PRIMARY KEY (rno),
	FOREIGN KEY (src)   REFERENCES Locations,
	FOREIGN KEY (dst)   REFERENCES Locations,
	FOREIGN KEY (offer) REFERENCES Drivers,
	FOREIGN KEY (assoc) REFERENCES Cars
);

CREATE TABLE Drivers(
	email CHAR(320) NOT NULL,
	licNo CHAR(16),
	FOREIGN KEY (email) REFERENCES Members
);

CREATE TABLE Cars(
	cno      CHAR(16),
	make     CHAR(16),
	model    CHAR(32),
	year     INTEGER,
	seats    INTEGER,
	gdate    DATE,
	register CHAR(320),
	PRIMARY KEY (cno),
	FOREIGN KEY (register) REFERENCES Drivers
):

CREATE TABLE Locations(
	lcode   CHAR(32),
	city    CHAR(64),
	prov    CHAR(64),
	address CHAR(128),
	PRIMARY KEY (lcode)
);

CREATE TABLE Requests(
	amount  REAL,
	qdate   DATE,
	pickup  CHAR(32),
	dropoff CHAR(32),
	FOREIGN KEY (pickup)  REFERENCES Locations,
	FOREIGN KEY (dropoff) REFERENCES Locations
);

CREATE TABLE Enroute(
	rno   CHAR(16),
	lcode CHAR(32),
	FOREIGN KEY (rno)   REFERENCES Rides,
	FOREIGN KEY (lcode) REFERENCES Locations
);
