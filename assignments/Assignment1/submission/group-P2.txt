CREATE TABLE Bookings(
	seats       INTEGER,
	bno         CHAR(16), 
	cost        REAL,
	rideBooked  CHAR(16) NOT NULL,
	dropOff     CHAR(16),
	pickUp      CHAR(16),
	reservedFor CHAR(320) NOT NULL,
	PRIMARY KEY (bno),
	FOREIGN KEY (rideBook)    REFERENCES Rides,
	FOREIGN KEY (dropOff)     REFERENCES Locations,
	FOREIGN KEY (pickup)      REFERENCES Locations,
	FOREIGN KEY (reservedFor) REFERENCES Members
);

CREATE TABLE Members(
	email CHAR(320),
	name  CHAR(64),
	phone CHAR(16),
	PRIMARY KEY (email)
);

CREATE TABLE Reviews(
	rid      CHAR(16),
	rdate    DATE,
	rtext    CHAR(320),
	rscore   INTEGER,
	reviewer CHAR(320) NOT NULL,
	reviewee CHAR(320) NOT NULL,
	assoc    CHAR(16),
	PRIMARY KEY (rid),
	FOREIGN KEY (reviewer) REFERENCES Members,
	FOREIGN KEY (reviewee) REFERENCES Members,
	FOREIGN KEY (assoc)    REFERENCES Rides
);

CREATE TABLE Rides(
	rno     CHAR(16),
	price   REAL,
	rdate   DATE,
	seats   INTEGER,
	lugDesc CHAR(1024),
	src     CHAR(32),
	dst     CHAR(32),
	offer   CHAR(320),
	assoc   CHAR(16),
	PRIMARY KEY (rno),
	FOREIGN KEY (src)   REFERENCES Locations,
	FOREIGN KEY (dst)   REFERENCES Locations,
	FOREIGN KEY (offer) REFERENCES Drivers,
	FOREIGN KEY (assoc) REFERENCES Cars
);
